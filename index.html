function loadLessons(month) { const listContainer = document.getElementById('lessonLinks'); listContainer.innerHTML = ''; lessonsData[month].forEach(lesson => { const btn = document.createElement('div'); btn.className = "flex justify-between items-center bg-slate-800 p-4 rounded-xl cursor-pointer hover:bg-slate-700 transition"; if(lesson.type === 'youtube') { btn.innerHTML = ${lesson.title}; btn.onclick = () => { document.getElementById('mainVideo').src = https://www.youtube.com/embed/${lesson.id}; window.scrollTo({ top: 0, behavior: 'smooth' }); }; } else { btn.innerHTML = ${lesson.title}; btn.onclick = () => window.open(lesson.url, '_blank'); } listContainer.appendChild(btn); }); } // Initialize with January loadLessons('january');
